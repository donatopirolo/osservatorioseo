# Tracker v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the empty tracker v1 with a working v2 that shows 5 sections of real Cloudflare Radar data for Italy and the World.

**Architecture:** Rewrite models, radar client, and collector to match tested API endpoints. Replace server-side SVG charts with a Jinja template that embeds JSON data and renders interactive charts via vanilla JS. The publisher method `_ssg_tracker` is updated to pass v2 snapshot data to the new template.

**Tech Stack:** Python 3.12, Pydantic v2, httpx (async), pytest + pytest-httpx, Jinja2, Tailwind CSS, vanilla JS (SVG charts in browser).

"""Pydantic models for the tracker v2 subsystem."""

from __future__ import annotations

from datetime import datetime

from pydantic import BaseModel, ConfigDict, Field


# --- Shared ---

class TimeseriesPoint(BaseModel):
    model_config = ConfigDict(extra="forbid")
    date: datetime
    value: float


# --- Section 1: Top 10 ---

class TopDomainEntry(BaseModel):
    model_config = ConfigDict(extra="forbid")
    rank: int = Field(ge=1)
    domain: str
    categories: list[str] = Field(default_factory=list)
    timeseries: list[TimeseriesPoint] = Field(default_factory=list)


# --- Section 2: AI Platforms ---

class AIPlatformEntry(BaseModel):
    model_config = ConfigDict(extra="forbid")
    domain: str
    label: str
    type: str  # "chatbot" | "ai_search" | "ai_tool"
    rank: int | None = None
    bucket: str  # "200", "500", ..., ">200000"


# --- Section 3: Bot vs Human ---

class BotHumanPoint(BaseModel):
    model_config = ConfigDict(extra="forbid")
    date: datetime
    human_pct: float
    bot_pct: float


class BotHumanTimeseries(BaseModel):
    model_config = ConfigDict(extra="forbid")
    points: list[BotHumanPoint] = Field(default_factory=list)


# --- Section 4a: AI Bots by user agent ---

class AIBotPoint(BaseModel):
    model_config = ConfigDict(extra="forbid")
    date: datetime
    values: dict[str, float]  # agent_name -> pct


class AIBotsTimeseries(BaseModel):
    model_config = ConfigDict(extra="forbid")
    agents: list[str] = Field(default_factory=list)
    points: list[AIBotPoint] = Field(default_factory=list)


# --- Section 4b: Crawl Purpose ---

class CrawlPurposePoint(BaseModel):
    model_config = ConfigDict(extra="forbid")
    date: datetime
    values: dict[str, float]  # purpose -> pct


class CrawlPurposeTimeseries(BaseModel):
    model_config = ConfigDict(extra="forbid")
    purposes: list[str] = Field(default_factory=list)
    points: list[CrawlPurposePoint] = Field(default_factory=list)


# --- Section 5: Industry ---

class IndustryEntry(BaseModel):
    model_config = ConfigDict(extra="forbid")
    industry: str
    pct: float


# --- Metadata ---

class SnapshotMetadata(BaseModel):
    model_config = ConfigDict(extra="forbid")
    radar_calls: int = 0
    warnings: list[str] = Field(default_factory=list)


# --- Snapshot ---

class TrackerSnapshot(BaseModel):
    model_config = ConfigDict(extra="forbid")

    schema_version: str = "2.0"
    year: int
    week: int = Field(ge=1, le=53)
    generated_at: datetime

    # Section 1
    top10_it: list[TopDomainEntry] = Field(default_factory=list)
    top10_global: list[TopDomainEntry] = Field(default_factory=list)

    # Section 2
    ai_platforms_it: list[AIPlatformEntry] = Field(default_factory=list)
    ai_platforms_global: list[AIPlatformEntry] = Field(default_factory=list)

    # Section 3
    bot_human_it: BotHumanTimeseries = Field(default_factory=BotHumanTimeseries)
    bot_human_global: BotHumanTimeseries = Field(default_factory=BotHumanTimeseries)

    # Section 4a
    ai_bots_ua_it: AIBotsTimeseries = Field(default_factory=AIBotsTimeseries)
    ai_bots_ua_global: AIBotsTimeseries = Field(default_factory=AIBotsTimeseries)

    # Section 4b
    crawl_purpose_it: CrawlPurposeTimeseries = Field(default_factory=CrawlPurposeTimeseries)
    crawl_purpose_global: CrawlPurposeTimeseries = Field(default_factory=CrawlPurposeTimeseries)

    # Section 5
    industry_it: list[IndustryEntry] = Field(default_factory=list)
    industry_global: list[IndustryEntry] = Field(default_factory=list)

    metadata: SnapshotMetadata


# --- Monthly Report (unchanged from v1) ---

class ReportTakeaway(BaseModel):
    model_config = ConfigDict(extra="forbid")
    title: str
    body: str


class TrackerMonthlyReport(BaseModel):
    model_config = ConfigDict(extra="forbid")

    schema_version: str = "2.0"
    year: int
    month: int = Field(ge=1, le=12)

    title_it: str
    subtitle_it: str
    hero_mover: str

    executive_summary: list[str] = Field(default_factory=list, max_length=6)
    narrative: str
    takeaways: list[ReportTakeaway] = Field(default_factory=list, max_length=8)
    outlook: str

    snapshot_week_refs: list[str] = Field(default_factory=list)
    generated_at: datetime
    model_used: str
    cost_eur: float = 0.0
```

"""Async client for Cloudflare Radar API (v2).

Docs: https://developers.cloudflare.com/api/resources/radar/
Free tier, requires API token with Account Analytics Read permission.
"""

from __future__ import annotations

import logging
from typing import Any

import httpx

logger = logging.getLogger(__name__)

RADAR_BASE_URL = "https://api.cloudflare.com/client/v4/radar"


class RadarClientError(Exception):
    """Raised on any non-2xx response or API-level failure."""


class RadarClient:
    """Async wrapper for Cloudflare Radar endpoints used by tracker v2."""

    def __init__(
        self,
        api_token: str,
        timeout_s: int = 30,
        base_url: str = RADAR_BASE_URL,
    ) -> None:
        self._api_token = api_token
        self._timeout = timeout_s
        self._base_url = base_url.rstrip("/")

    # --- Section 1: Ranking ---

    async def ranking_top(
        self,
        *,
        location: str | None = None,
        limit: int = 10,
        ranking_type: str = "POPULAR",
    ) -> list[dict[str, Any]]:
        """Top N domains. Returns list of {rank, domain, categories}."""
        params: dict[str, Any] = {"limit": limit, "rankingType": ranking_type}
        if location:
            params["location"] = location
        data = await self._get("/ranking/top", params)
        rows = data.get("result", {}).get("top_0", [])
        return [
            {
                "rank": row["rank"],
                "domain": row["domain"],
                "categories": [c["name"] for c in row.get("categories", [])],
            }
            for row in rows
        ]

    async def domain_detail(
        self,
        *,
        domain: str,
        location: str | None = None,
    ) -> dict[str, Any]:
        """Single domain rank + bucket. Returns {rank, bucket, categories}."""
        params: dict[str, Any] = {}
        if location:
            params["location"] = location
        data = await self._get(f"/ranking/domain/{domain}", params)
        det = data.get("result", {}).get("details_0", {})
        return {
            "rank": det.get("rank"),
            "bucket": str(det.get("bucket", "?")),
        }

    async def domain_timeseries(
        self,
        *,
        domain: str,
        location: str | None = None,
        date_range: str = "52w",
    ) -> list[dict[str, Any]]:
        """Ranking timeseries for a single domain. Returns [{date, rank}]."""
        params: dict[str, Any] = {"domains": domain, "dateRange": date_range}
        if location:
            params["location"] = location
        data = await self._get("/ranking/timeseries_groups", params)
        serie = data.get("result", {}).get("serie_0", {})
        timestamps = serie.get("timestamps", [])
        ranks = serie.get(domain, [])
        return [
            {"date": ts, "rank": int(r)}
            for ts, r in zip(timestamps, ranks, strict=False)
        ]

    # --- Section 3: Bot vs Human ---

    async def bot_human_timeseries(
        self,
        *,
        location: str | None = None,
        date_range: str = "12w",
    ) -> list[dict[str, float]]:
        """Bot vs human %. Returns [{date, human_pct, bot_pct}]."""
        params: dict[str, Any] = {"dateRange": date_range}
        if location:
            params["location"] = location
        data = await self._get("/http/timeseries_groups/bot_class", params)
        serie = data.get("result", {}).get("serie_0", {})
        timestamps = serie.get("timestamps", [])
        humans = serie.get("human", [])
        bots = serie.get("bot", [])
        return [
            {"date": ts, "human_pct": float(h), "bot_pct": float(b)}
            for ts, h, b in zip(timestamps, humans, bots, strict=False)
        ]

    # --- Section 4a: AI Bots ---

    async def ai_bots_user_agent(
        self,
        *,
        location: str | None = None,
        date_range: str = "12w",
        agg_interval: str = "1w",
    ) -> tuple[list[str], list[dict[str, Any]]]:
        """AI bot timeseries by user agent. Returns (agents, [{date, values}])."""
        params: dict[str, Any] = {"dateRange": date_range, "aggInterval": agg_interval}
        if location:
            params["location"] = location
        data = await self._get("/ai/bots/timeseries_groups/user_agent", params)
        serie = data.get("result", {}).get("serie_0", {})
        timestamps = serie.get("timestamps", [])
        agents = [k for k in serie if k != "timestamps"]
        points = []
        for i, ts in enumerate(timestamps):
            values = {agent: float(serie[agent][i]) for agent in agents}
            points.append({"date": ts, "values": values})
        return agents, points

    # --- Section 4b: Crawl Purpose ---

    async def crawl_purpose(
        self,
        *,
        location: str | None = None,
        date_range: str = "12w",
        agg_interval: str = "1w",
    ) -> tuple[list[str], list[dict[str, Any]]]:
        """Crawl purpose timeseries. Returns (purposes, [{date, values}])."""
        params: dict[str, Any] = {"dateRange": date_range, "aggInterval": agg_interval}
        if location:
            params["location"] = location
        data = await self._get("/ai/bots/timeseries_groups/crawl_purpose", params)
        serie = data.get("result", {}).get("serie_0", {})
        timestamps = serie.get("timestamps", [])
        purposes = [k for k in serie if k != "timestamps"]
        points = []
        for i, ts in enumerate(timestamps):
            values = {p: float(serie[p][i]) for p in purposes}
            points.append({"date": ts, "values": values})
        return purposes, points

    # --- Section 5: Industry ---

    async def industry_summary(
        self,
        *,
        location: str | None = None,
        date_range: str = "28d",
    ) -> list[dict[str, Any]]:
        """AI bot crawl by industry. Returns [{industry, pct}] sorted desc."""
        params: dict[str, Any] = {"dateRange": date_range}
        if location:
            params["location"] = location
        data = await self._get("/ai/bots/summary/industry", params)
        summary = data.get("result", {}).get("summary_0", {})
        entries = [
            {"industry": k, "pct": float(v)}
            for k, v in summary.items()
        ]
        entries.sort(key=lambda e: -e["pct"] if e["industry"] != "other" else 0)
        return entries

    # --- Internal ---

    async def _get(self, path: str, params: dict[str, Any]) -> dict:
        url = f"{self._base_url}{path}"
        headers = {"Authorization": f"Bearer {self._api_token}"}
        async with httpx.AsyncClient(timeout=self._timeout) as client:
            resp = await client.get(url, params=params, headers=headers)
        if resp.status_code >= 400:
            raise RadarClientError(
                f"Radar API error {resp.status_code}: {resp.text[:200]}"
            )
        data = resp.json()
        if not data.get("success", False):
            errs = data.get("errors", [])
            msg = "; ".join(e.get("message", str(e)) for e in errs) or "unknown"
            raise RadarClientError(f"Radar API reported failure: {msg}")
        return data
```

"""Collector v2: orchestrates Radar API into a TrackerSnapshot.

Fetches data for both Italy (location=IT) and Global (no location filter),
building a complete snapshot with 5 sections × 2 locations.
"""

from __future__ import annotations

import logging
from datetime import UTC, datetime
from pathlib import Path

import yaml

from osservatorio_seo.tracker.models import (
    AIBotPoint,
    AIBotsTimeseries,
    AIPlatformEntry,
    BotHumanPoint,
    BotHumanTimeseries,
    CrawlPurposePoint,
    CrawlPurposeTimeseries,
    IndustryEntry,
    SnapshotMetadata,
    TimeseriesPoint,
    TopDomainEntry,
    TrackerSnapshot,
)
from osservatorio_seo.tracker.radar_client import RadarClient

logger = logging.getLogger(__name__)

_DEFAULT_PLATFORMS = Path(__file__).resolve().parents[3] / "config" / "tracker_platforms.yaml"


class TrackerCollector:
    """Orchestrates Radar API calls and produces a TrackerSnapshot."""

    def __init__(
        self,
        radar: RadarClient,
        platforms_config: Path = _DEFAULT_PLATFORMS,
    ) -> None:
        self._radar = radar
        self._platforms = self._load_platforms(platforms_config)
        self._calls = 0
        self._warnings: list[str] = []

    @staticmethod
    def _load_platforms(path: Path) -> list[dict]:
        with open(path, encoding="utf-8") as f:
            return yaml.safe_load(f)["platforms"]

    async def collect(self, year: int, week: int) -> TrackerSnapshot:
        self._calls = 0
        self._warnings = []

        top10_it = await self._fetch_top10(location="IT")
        top10_global = await self._fetch_top10(location=None)

        ai_it = await self._fetch_ai_platforms(location="IT")
        ai_global = await self._fetch_ai_platforms(location=None)

        bh_it = await self._safe(self._fetch_bot_human, "bot_human_it", location="IT")
        bh_global = await self._safe(self._fetch_bot_human, "bot_human_global", location=None)

        bots_it = await self._safe(self._fetch_ai_bots_ua, "ai_bots_ua_it", location="IT")
        bots_global = await self._safe(self._fetch_ai_bots_ua, "ai_bots_ua_global", location=None)

        cp_it = await self._safe(self._fetch_crawl_purpose, "crawl_purpose_it", location="IT")
        cp_global = await self._safe(self._fetch_crawl_purpose, "crawl_purpose_global", location=None)

        ind_it = await self._safe(self._fetch_industry, "industry_it", location="IT")
        ind_global = await self._safe(self._fetch_industry, "industry_global", location=None)

        return TrackerSnapshot(
            year=year,
            week=week,
            generated_at=datetime.now(UTC),
            top10_it=top10_it,
            top10_global=top10_global,
            ai_platforms_it=ai_it,
            ai_platforms_global=ai_global,
            bot_human_it=bh_it or BotHumanTimeseries(),
            bot_human_global=bh_global or BotHumanTimeseries(),
            ai_bots_ua_it=bots_it or AIBotsTimeseries(),
            ai_bots_ua_global=bots_global or AIBotsTimeseries(),
            crawl_purpose_it=cp_it or CrawlPurposeTimeseries(),
            crawl_purpose_global=cp_global or CrawlPurposeTimeseries(),
            industry_it=ind_it or [],
            industry_global=ind_global or [],
            metadata=SnapshotMetadata(
                radar_calls=self._calls,
                warnings=list(self._warnings),
            ),
        )

    async def _safe(self, fn, label: str, **kwargs):
        try:
            return await fn(**kwargs)
        except Exception as e:  # noqa: BLE001
            self._warnings.append(f"{label}: {e}")
            logger.warning("%s failed: %s", label, e)
            return None

    async def _fetch_top10(self, location: str | None) -> list[TopDomainEntry]:
        self._calls += 1
        rows = await self._radar.ranking_top(location=location, limit=10)
        entries = []
        for row in rows:
            self._calls += 1
            try:
                ts_points = await self._radar.domain_timeseries(
                    domain=row["domain"],
                    location=location,
                    date_range="52w",
                )
                timeseries = [
                    TimeseriesPoint(date=datetime.fromisoformat(p["date"].replace("Z", "+00:00")), value=float(p["rank"]))
                    for p in ts_points
                ]
            except Exception as e:  # noqa: BLE001
                self._warnings.append(f"timeseries({row['domain']}): {e}")
                timeseries = []
            entries.append(
                TopDomainEntry(
                    rank=row["rank"],
                    domain=row["domain"],
                    categories=row.get("categories", []),
                    timeseries=timeseries,
                )
            )
        return entries

    async def _fetch_ai_platforms(self, location: str | None) -> list[AIPlatformEntry]:
        entries = []
        for p in self._platforms:
            self._calls += 1
            try:
                detail = await self._radar.domain_detail(
                    domain=p["domain"], location=location
                )
            except Exception as e:  # noqa: BLE001
                self._warnings.append(f"domain_detail({p['domain']}): {e}")
                detail = {"rank": None, "bucket": "?"}
            entries.append(
                AIPlatformEntry(
                    domain=p["domain"],
                    label=p["label"],
                    type=p["type"],
                    rank=detail.get("rank"),
                    bucket=detail.get("bucket", "?"),
                )
            )
        return entries

    async def _fetch_bot_human(self, location: str | None) -> BotHumanTimeseries:
        self._calls += 1
        raw = await self._radar.bot_human_timeseries(location=location, date_range="12w")
        points = [
            BotHumanPoint(
                date=datetime.fromisoformat(p["date"].replace("Z", "+00:00")),
                human_pct=p["human_pct"],
                bot_pct=p["bot_pct"],
            )
            for p in raw
        ]
        return BotHumanTimeseries(points=points)

    async def _fetch_ai_bots_ua(self, location: str | None) -> AIBotsTimeseries:
        self._calls += 1
        agents, raw = await self._radar.ai_bots_user_agent(location=location)
        points = [
            AIBotPoint(
                date=datetime.fromisoformat(p["date"].replace("Z", "+00:00")),
                values=p["values"],
            )
            for p in raw
        ]
        return AIBotsTimeseries(agents=agents, points=points)

    async def _fetch_crawl_purpose(self, location: str | None) -> CrawlPurposeTimeseries:
        self._calls += 1
        purposes, raw = await self._radar.crawl_purpose(location=location)
        points = [
            CrawlPurposePoint(
                date=datetime.fromisoformat(p["date"].replace("Z", "+00:00")),
                values=p["values"],
            )
            for p in raw
        ]
        return CrawlPurposeTimeseries(purposes=purposes, points=points)

    async def _fetch_industry(self, location: str | None) -> list[IndustryEntry]:
        self._calls += 1
        raw = await self._radar.industry_summary(location=location)
        return [IndustryEntry(industry=e["industry"], pct=e["pct"]) for e in raw]

    @staticmethod
    def persist(snapshot: TrackerSnapshot, base_dir: Path) -> Path:
        snapshots_dir = Path(base_dir) / "snapshots"
        snapshots_dir.mkdir(parents=True, exist_ok=True)
        filename = f"{snapshot.year}-W{snapshot.week:02d}.json"
        target = snapshots_dir / filename
        target.write_text(snapshot.model_dump_json(indent=2) + "\n", encoding="utf-8")
        return target

"""Weekly tracker update — fetches Radar data, saves snapshot.

Usage:
    python scripts/update_tracker.py

Environment variables:
    CLOUDFLARE_RADAR_TOKEN     Required, token with Account Analytics Read
    CLOUDFLARE_API_TOKEN       Fallback if CLOUDFLARE_RADAR_TOKEN not set

import asyncio
import os
import sys
from datetime import date
from pathlib import Path

"use strict";

(function () {
  const DATA = window.__TRACKER_DATA__;
  if (!DATA) return;

  let currentView = localStorage.getItem("tracker_view") || "it";

  // --- Utils ---
  function $(sel, ctx) { return (ctx || document).querySelector(sel); }
  function $$(sel, ctx) { return [...(ctx || document).querySelectorAll(sel)]; }

  const COLORS = [
    "#00f63e", "#f5a623", "#e24c4c", "#4ca6e2", "#e24ce2",
    "#4ce2c4", "#e2e24c", "#a64cf5", "#f54c8a", "#8af54c",
  ];
  const GREEN = "#00f63e";
  const GREY = "#919191";

  // --- SVG helpers ---
  function svgEl(tag, attrs) {
    const el = document.createElementNS("http://www.w3.org/2000/svg", tag);
    for (const [k, v] of Object.entries(attrs || {})) el.setAttribute(k, v);
    return el;
  }

  function clearChart(id) {
    const el = document.getElementById(id);
    if (el) el.innerHTML = "";
    return el;
  }

  // --- Section 1: Top 10 line chart ---
  function renderTop10(containerId, entries) {
    const wrap = clearChart(containerId);
    if (!wrap || !entries.length) return;

    const W = 700, H = 350, PAD = { t: 20, r: 20, b: 40, l: 50 };
    const plotW = W - PAD.l - PAD.r, plotH = H - PAD.t - PAD.b;

    // Collect all dates and rank range
    let allDates = new Set();
    for (const e of entries) for (const p of e.timeseries) allDates.add(p.date);
    allDates = [...allDates].sort();
    if (!allDates.length) { wrap.textContent = "Dati insufficienti"; return; }

    const maxRank = Math.max(
      ...entries.flatMap(e => e.timeseries.map(p => p.value)),
      10
    );

    const xScale = (i) => PAD.l + (i / Math.max(allDates.length - 1, 1)) * plotW;
    const yScale = (rank) => PAD.t + ((rank - 1) / (maxRank - 1)) * plotH;

    const svg = svgEl("svg", { viewBox: `0 0 ${W} ${H}`, class: "w-full" });

    // Y axis labels
    for (const r of [1, Math.round(maxRank / 2), maxRank]) {
      const y = yScale(r);
      svg.appendChild(svgEl("line", { x1: PAD.l, y1: y, x2: W - PAD.r, y2: y, stroke: "#333", "stroke-width": 0.5 }));
      const txt = svgEl("text", { x: PAD.l - 8, y: y + 4, fill: GREY, "font-size": 11, "text-anchor": "end" });
      txt.textContent = `#${r}`;
      svg.appendChild(txt);
    }

    // X axis labels (first, mid, last)
    for (const idx of [0, Math.floor(allDates.length / 2), allDates.length - 1]) {
      const txt = svgEl("text", { x: xScale(idx), y: H - 5, fill: GREY, "font-size": 10, "text-anchor": "middle" });
      txt.textContent = allDates[idx].slice(0, 10);
      svg.appendChild(txt);
    }

    // Lines
    const dateIdx = Object.fromEntries(allDates.map((d, i) => [d, i]));
    const hidden = new Set();

    function drawLines() {
      $$(".chart-line", svg).forEach(el => el.remove());
      entries.forEach((entry, ei) => {
        if (hidden.has(entry.domain)) return;
        const pts = entry.timeseries
          .filter(p => dateIdx[p.date] !== undefined)
          .map(p => `${xScale(dateIdx[p.date])},${yScale(p.value)}`);
        if (pts.length < 2) return;
        svg.appendChild(svgEl("polyline", {
          points: pts.join(" "),
          fill: "none",
          stroke: COLORS[ei % COLORS.length],
          "stroke-width": 2,
          class: "chart-line",
        }));
      });
    }

    drawLines();
    wrap.appendChild(svg);

    // Legend
    const legend = document.createElement("div");
    legend.className = "flex flex-wrap gap-3 mt-3 text-sm";
    entries.forEach((entry, ei) => {
      const btn = document.createElement("button");
      btn.className = "flex items-center gap-1 opacity-100 transition-opacity";
      btn.innerHTML = `<span class="inline-block w-3 h-3 rounded-full" style="background:${COLORS[ei % COLORS.length]}"></span>${entry.domain}`;
      btn.addEventListener("click", () => {
        if (hidden.has(entry.domain)) {
          hidden.delete(entry.domain);
          btn.style.opacity = "1";
        } else {
          hidden.add(entry.domain);
          btn.style.opacity = "0.3";
        }
        drawLines();
      });
      legend.appendChild(btn);
    });
    wrap.appendChild(legend);
  }

  // --- Section 2: AI platforms table ---
  function renderPlatforms(containerId, platformsIt, platformsGlobal) {
    const wrap = clearChart(containerId);
    if (!wrap) return;
    const platforms = currentView === "it" ? platformsIt : platformsGlobal;
    const other = currentView === "it" ? platformsGlobal : platformsIt;

    const table = document.createElement("table");
    table.className = "w-full text-sm border-collapse";
    table.innerHTML = `<thead><tr class="text-left border-b border-[var(--outline-variant)]">
      <th class="py-2">Dominio</th><th>Tipo</th>
      <th class="text-right">Rank</th><th class="text-right">Bucket</th>
      <th class="text-right">${currentView === "it" ? "Mondo" : "Italia"}</th>
    </tr></thead>`;
    const tbody = document.createElement("tbody");
    platforms.forEach((p, i) => {
      const otherP = other[i];
      const tr = document.createElement("tr");
      tr.className = "border-b border-[var(--outline-variant)]/30";
      tr.innerHTML = `<td class="py-1.5 font-mono">${p.domain}</td>
        <td class="text-[var(--outline)]">${p.type}</td>
        <td class="text-right font-mono">${p.rank ? "#" + p.rank : "-"}</td>
        <td class="text-right font-mono">${p.bucket.startsWith(">") ? p.bucket : "top " + p.bucket}</td>
        <td class="text-right font-mono text-[var(--outline)]">${otherP ? (otherP.bucket.startsWith(">") ? otherP.bucket : "top " + otherP.bucket) : "-"}</td>`;
      tbody.appendChild(tr);
    });
    table.appendChild(tbody);
    wrap.appendChild(table);
  }

  // --- Section 3: Bot vs Human area chart ---
  function renderBotHuman(containerId, data) {
    const wrap = clearChart(containerId);
    if (!wrap || !data.points.length) return;

    const W = 700, H = 250, PAD = { t: 20, r: 20, b: 40, l: 50 };
    const plotW = W - PAD.l - PAD.r, plotH = H - PAD.t - PAD.b;
    const n = data.points.length;

    const xScale = (i) => PAD.l + (i / Math.max(n - 1, 1)) * plotW;
    const yScale = (pct) => PAD.t + ((100 - pct) / 100) * plotH;

    const svg = svgEl("svg", { viewBox: `0 0 ${W} ${H}`, class: "w-full" });

    // Bot area (bottom)
    let botPath = `M${xScale(0)},${yScale(0)}`;
    data.points.forEach((p, i) => { botPath += ` L${xScale(i)},${yScale(p.bot_pct)}`; });
    botPath += ` L${xScale(n - 1)},${yScale(0)} Z`;
    svg.appendChild(svgEl("path", { d: botPath, fill: "#f5a623", "fill-opacity": "0.3", stroke: "#f5a623", "stroke-width": 1.5 }));

    // Human line at top
    let humanPts = data.points.map((p, i) => `${xScale(i)},${yScale(100)}`);
    // Actually draw bot line (human is just 100 - bot)
    let botLinePts = data.points.map((p, i) => `${xScale(i)},${yScale(p.bot_pct)}`).join(" ");
    svg.appendChild(svgEl("polyline", { points: botLinePts, fill: "none", stroke: "#f5a623", "stroke-width": 2 }));

    // Labels
    const first = data.points[0], last = data.points[n - 1];
    for (const [idx, p] of [[0, first], [n - 1, last]]) {
      const txt = svgEl("text", { x: xScale(idx), y: H - 5, fill: GREY, "font-size": 10, "text-anchor": idx === 0 ? "start" : "end" });
      txt.textContent = p.date.slice(0, 10);
      svg.appendChild(txt);
    }

    // Y axis
    for (const pct of [0, 10, 20, 30, 40]) {
      const y = yScale(pct);
      svg.appendChild(svgEl("line", { x1: PAD.l, y1: y, x2: W - PAD.r, y2: y, stroke: "#333", "stroke-width": 0.5 }));
      const txt = svgEl("text", { x: PAD.l - 8, y: y + 4, fill: GREY, "font-size": 11, "text-anchor": "end" });
      txt.textContent = `${pct}%`;
      svg.appendChild(txt);
    }

    // Current value annotation
    const lastBot = last.bot_pct;
    const ann = svgEl("text", { x: xScale(n - 1) + 5, y: yScale(lastBot), fill: "#f5a623", "font-size": 13, "font-weight": "bold" });
    ann.textContent = `${lastBot.toFixed(1)}% bot`;
    svg.appendChild(ann);

    wrap.appendChild(svg);
  }

  // --- Section 4a: AI bots lines ---
  function renderAIBots(containerId, data) {
    const wrap = clearChart(containerId);
    if (!wrap || !data.points.length) return;

    const entries = data.agents.map((agent, i) => ({
      domain: agent,
      timeseries: data.points.map(p => ({ date: p.date, value: p.values[agent] || 0 })),
    }));

    // Reuse line chart logic
    const W = 700, H = 300, PAD = { t: 20, r: 20, b: 40, l: 50 };
    const plotW = W - PAD.l - PAD.r, plotH = H - PAD.t - PAD.b;

    const allDates = data.points.map(p => p.date);
    const maxVal = Math.max(...data.points.flatMap(p => Object.values(p.values)), 10);

    const xScale = (i) => PAD.l + (i / Math.max(allDates.length - 1, 1)) * plotW;
    const yScale = (v) => PAD.t + ((maxVal - v) / maxVal) * plotH;

    const svg = svgEl("svg", { viewBox: `0 0 ${W} ${H}`, class: "w-full" });

    // Grid
    for (const v of [0, Math.round(maxVal / 2), maxVal]) {
      const y = yScale(v);
      svg.appendChild(svgEl("line", { x1: PAD.l, y1: y, x2: W - PAD.r, y2: y, stroke: "#333", "stroke-width": 0.5 }));
      const txt = svgEl("text", { x: PAD.l - 8, y: y + 4, fill: GREY, "font-size": 11, "text-anchor": "end" });
      txt.textContent = `${v}%`;
      svg.appendChild(txt);
    }

    const hidden = new Set();
    function drawLines() {
      $$(".chart-line", svg).forEach(el => el.remove());
      entries.forEach((entry, ei) => {
        if (hidden.has(entry.domain)) return;
        const pts = entry.timeseries.map((p, i) => `${xScale(i)},${yScale(p.value)}`);

          points: pts.join(" "), fill: "none",

    wrap.appendChild(svg);

    legend.className = "flex flex-wrap gap-3 mt-3 text-sm";

  }

  // --- Section 4b: Crawl purpose stacked area ---
  function renderCrawlPurpose(containerId, data) {
    const wrap = clearChart(containerId);
    if (!wrap || !data.points.length) return;

    const W = 700, H = 250, PAD = { t: 20, r: 20, b: 40, l: 50 };
    const plotW = W - PAD.l - PAD.r, plotH = H - PAD.t - PAD.b;
    const n = data.points.length;
    const purposes = data.purposes;
    const purposeColors = { "User Action": "#00f63e", "Training": "#e24c4c", "Mixed Purpose": "#f5a623", "Search": "#4ca6e2", "Undeclared": "#919191" };

    const xScale = (i) => PAD.l + (i / Math.max(n - 1, 1)) * plotW;
    const yScale = (pct) => PAD.t + ((100 - pct) / 100) * plotH;

    const svg = svgEl("svg", { viewBox: `0 0 ${W} ${H}`, class: "w-full" });

    // Stack from bottom
    const stacked = data.points.map(p => {
      let cum = 0;
      return purposes.map(pur => { cum += p.values[pur] || 0; return cum; });
    });

    // Draw areas bottom-up
    for (let pi = purposes.length - 1; pi >= 0; pi--) {
      const top = stacked.map((s, i) => `${xScale(i)},${yScale(s[pi])}`).join(" ");
      const bottom = pi > 0
        ? stacked.map((s, i) => `${xScale(i)},${yScale(s[pi - 1])}`).reverse().join(" ")
        : stacked.map((_, i) => `${xScale(n - 1 - i)},${yScale(0)}`).join(" ");
      svg.appendChild(svgEl("polygon", {
        points: `${top} ${bottom}`,
        fill: purposeColors[purposes[pi]] || COLORS[pi],
        "fill-opacity": "0.6",
      }));
    }

    // X labels
    for (const idx of [0, n - 1]) {
      const txt = svgEl("text", { x: xScale(idx), y: H - 5, fill: GREY, "font-size": 10, "text-anchor": idx === 0 ? "start" : "end" });
      txt.textContent = data.points[idx].date.slice(0, 10);
      svg.appendChild(txt);
    }

    wrap.appendChild(svg);

    // Legend
    const legend = document.createElement("div");
    legend.className = "flex flex-wrap gap-3 mt-3 text-sm";
    purposes.forEach((p, i) => {
      const span = document.createElement("span");
      span.className = "flex items-center gap-1";
      span.innerHTML = `<span class="inline-block w-3 h-3 rounded-full" style="background:${purposeColors[p] || COLORS[i]}"></span>${p}`;
      legend.appendChild(span);
    });
    wrap.appendChild(legend);
  }

  // --- Section 5: Industry bar chart ---
  function renderIndustry(containerId, entries) {
    const wrap = clearChart(containerId);
    if (!wrap || !entries.length) return;

    const top = entries.filter(e => e.industry !== "other").slice(0, 10);
    const barH = 28, gap = 4, W = 700, PAD = { l: 200, r: 60, t: 10, b: 10 };
    const H = PAD.t + top.length * (barH + gap) + PAD.b;
    const maxPct = Math.max(...top.map(e => e.pct));

    const svg = svgEl("svg", { viewBox: `0 0 ${W} ${H}`, class: "w-full" });

    top.forEach((entry, i) => {
      const y = PAD.t + i * (barH + gap);
      const barW = ((entry.pct / maxPct) * (W - PAD.l - PAD.r));

      // Label
      const label = svgEl("text", { x: PAD.l - 10, y: y + barH / 2 + 4, fill: "#e2e2e2", "font-size": 12, "text-anchor": "end" });
      label.textContent = entry.industry;
      svg.appendChild(label);

      // Bar
      svg.appendChild(svgEl("rect", { x: PAD.l, y: y, width: barW, height: barH, fill: GREEN, "fill-opacity": "0.7", rx: 3 }));

      // Value
      const val = svgEl("text", { x: PAD.l + barW + 6, y: y + barH / 2 + 4, fill: GREY, "font-size": 11 });
      val.textContent = `${entry.pct.toFixed(1)}%`;
      svg.appendChild(val);
    });

    wrap.appendChild(svg);
  }

  // --- Main render ---
  function renderAll() {
    const suffix = currentView === "it" ? "it" : "global";
    renderTop10("chart-top10", DATA[`top10_${suffix}`]);
    renderPlatforms("chart-platforms", DATA.ai_platforms_it, DATA.ai_platforms_global);
    renderBotHuman("chart-bot-human", DATA[`bot_human_${suffix}`]);
    renderAIBots("chart-ai-bots", DATA[`ai_bots_ua_${suffix}`]);
    renderCrawlPurpose("chart-crawl-purpose", DATA[`crawl_purpose_${suffix}`]);
    renderIndustry("chart-industry", DATA[`industry_${suffix}`]);
  }

  // --- Toggle ---
  document.addEventListener("DOMContentLoaded", () => {
    renderAll();

    $$("[data-view-toggle]").forEach(btn => {
      if (btn.dataset.viewToggle === currentView) btn.classList.add("active");
      btn.addEventListener("click", () => {
        currentView = btn.dataset.viewToggle;
        localStorage.setItem("tracker_view", currentView);
        $$("[data-view-toggle]").forEach(b => b.classList.remove("active"));
        btn.classList.add("active");
        renderAll();
      });
    });
  });
})();
